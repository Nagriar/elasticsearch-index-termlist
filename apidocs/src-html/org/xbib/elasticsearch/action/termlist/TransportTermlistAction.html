010import org.elasticsearch.ElasticSearchException;
011import org.elasticsearch.action.ShardOperationFailedException;
012import org.elasticsearch.action.support.DefaultShardOperationFailedException;
013import org.elasticsearch.action.support.broadcast.BroadcastShardOperationFailedException;
014import org.elasticsearch.action.support.broadcast.TransportBroadcastOperationAction;

034

037/**
038 * Termlist index/indices action.
039 */
040public class TransportTermlistAction
041        extends TransportBroadcastOperationAction<TermlistRequest, TermlistResponse, ShardTermlistRequest, ShardTermlistResponse> {
042
043    private final IndicesService indicesService;
044    private final Object termlistMutex = new Object();
045
046    @Inject
047    public TransportTermlistAction(Settings settings, ThreadPool threadPool, ClusterService clusterService,
048                                   TransportService transportService, IndicesService indicesService) {
049        super(settings, threadPool, clusterService, transportService);
050        this.indicesService = indicesService;
051    }
052
053    @Override
054    protected String executor() {
055        return ThreadPool.Names.MERGE;
056    }
057
058    @Override
059    protected String transportAction() {
060        return TermlistAction.NAME;
061    }
062
063    @Override
064    protected TermlistRequest newRequest() {
065        return new TermlistRequest();
066    }
067
068    @Override
069    protected TermlistResponse newResponse(TermlistRequest request, AtomicReferenceArray shardsResponses, ClusterState clusterState) {
070        int successfulShards = 0;
071        int failedShards = 0;
072        List<ShardOperationFailedException> shardFailures = null;
073        Set<String> termlist = new CompactHashSet<String>();
074        for (int i = 0; i < shardsResponses.length(); i++) {
075            Object shardResponse = shardsResponses.get(i);
076            if (shardResponse == null) {
077                // a non active shard, ignore...
078            } else if (shardResponse instanceof BroadcastShardOperationFailedException) {
079                failedShards++;
080                if (shardFailures == null) {
081                    shardFailures = newArrayList();
082                }
083                shardFailures.add(new DefaultShardOperationFailedException((BroadcastShardOperationFailedException) shardResponse));
084            } else {
085                successfulShards++;
086                if (shardResponse instanceof ShardTermlistResponse) {
087                    ShardTermlistResponse resp = (ShardTermlistResponse) shardResponse;
088                    termlist.addAll(resp.getTermList());
089                }
090            }
091        }
092        return new TermlistResponse(shardsResponses.length(), successfulShards, failedShards, shardFailures, termlist);
093    }
094
095    @Override
096    protected ShardTermlistRequest newShardRequest() {
097        return new ShardTermlistRequest();
098    }
099
100    @Override
101    protected ShardTermlistRequest newShardRequest(ShardRouting shard, TermlistRequest request) {
102        return new ShardTermlistRequest(shard.index(), shard.id(), request);
103    }
104
105    @Override
106    protected ShardTermlistResponse newShardResponse() {
107        return new ShardTermlistResponse();
108    }
109
110    @Override
111    protected ShardTermlistResponse shardOperation(ShardTermlistRequest request) throws ElasticSearchException {
112        synchronized (termlistMutex) {
113            InternalIndexShard indexShard = (InternalIndexShard) indicesService.indexServiceSafe(request.index()).shardSafe(request.shardId());
114            indexShard.store().directory();
115            Engine.Searcher searcher = indexShard.acquireSearcher();
116            try {
117                Set<String> set = new CompactHashSet<String>();
118                Fields fields = MultiFields.getFields(searcher.reader());
119                if (fields != null) {
120                    for (Iterator<String> it = fields.iterator(); it.hasNext(); ) {
121                        String field = it.next();
122                        if (field.charAt(0) == '_') {
123                            continue;
124                        }
125                        if (request.getField() == null || field.equals(request.getField())) {
126                            Terms terms = fields.terms(field);
127                            if (terms != null) {
128                                TermsEnum termsEnum = terms.iterator(null);
129                                BytesRef text;
130                                while ((text = termsEnum.next()) != null) {
131                                    set.add(text.utf8ToString());
132                                    System.out.println("field=" + field + "; text=" + text.utf8ToString());
133                                }
134                            }
135                        }
136                    }
137                }
138                return new ShardTermlistResponse(request.index(), request.shardId(), set);
139            } catch (IOException ex) {
140                throw new ElasticSearchException(ex.getMessage(), ex);
141            }
142        }
143    }
144
145    /**
146     * The termlist request works against primary shards.
147     */
148    @Override
149    protected GroupShardsIterator shards(ClusterState clusterState, TermlistRequest request, String[] concreteIndices) {
150        return clusterState.routingTable().activePrimaryShardsGrouped(concreteIndices, true);
151    }
152
153    @Override
154    protected ClusterBlockException checkGlobalBlock(ClusterState state, TermlistRequest request) {
155        return state.blocks().globalBlockedException(ClusterBlockLevel.METADATA);
156    }
157
158    @Override
159    protected ClusterBlockException checkRequestBlock(ClusterState state, TermlistRequest request, String[] concreteIndices) {
160        return state.blocks().indicesBlockedException(ClusterBlockLevel.METADATA, concreteIndices);
161    }
162}

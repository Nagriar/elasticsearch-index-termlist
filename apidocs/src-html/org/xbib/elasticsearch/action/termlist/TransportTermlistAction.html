040    /**
041     * Termlist index/indices action.
042     */
043    public class TransportTermlistAction
044            extends TransportBroadcastOperationAction<TermlistRequest, TermlistResponse, ShardTermlistRequest, ShardTermlistResponse> {
045    
046        private final IndicesService indicesService;
047    
048        @Inject
049        public TransportTermlistAction(Settings settings, ThreadPool threadPool, ClusterService clusterService,
050                                       TransportService transportService, IndicesService indicesService) {
051            super(settings, threadPool, clusterService, transportService);
052            this.indicesService = indicesService;
053        }
054    
055        @Override
056        protected String executor() {
057            return ThreadPool.Names.GENERIC;
058        }
059    
060        @Override
061        protected String transportAction() {
062            return TermlistAction.NAME;
063        }
064    
065        @Override
066        protected TermlistRequest newRequest() {
067            return new TermlistRequest();
068        }
069    
070        @Override
071        protected TermlistResponse newResponse(TermlistRequest request, AtomicReferenceArray shardsResponses, ClusterState clusterState) {
072            int successfulShards = 0;
073            int failedShards = 0;
074            List<ShardOperationFailedException> shardFailures = null;
075            Map<String, TermInfo> map = new CompactHashMap<String,TermInfo>();
076            for (int i = 0; i < shardsResponses.length(); i++) {
077                Object shardResponse = shardsResponses.get(i);
078                if (shardResponse instanceof BroadcastShardOperationFailedException) {
079                    failedShards++;
080                    if (shardFailures == null) {
081                        shardFailures = newLinkedList();
082                    }
083                    shardFailures.add(new DefaultShardOperationFailedException((BroadcastShardOperationFailedException) shardResponse));
084                } else {
085                    successfulShards++;
086                    if (shardResponse instanceof ShardTermlistResponse) {
087                        ShardTermlistResponse resp = (ShardTermlistResponse) shardResponse;
088                        merge(map, resp.getTermList());
089                    }
090                }
091            }
092            map = request.getWithTotalFreq() ? sortTotalFreq(map, request.getSize()) :
093                    request.getWithDocFreq() ? sortDocFreq(map, request.getSize()) :
094                            truncate(map, request.getSize());
095    
096            return new TermlistResponse(shardsResponses.length(), successfulShards, failedShards, shardFailures, map);
097        }

132        @Override
133        protected ShardTermlistResponse shardOperation(ShardTermlistRequest request) throws ElasticsearchException {
134            InternalIndexShard indexShard = (InternalIndexShard) indicesService.indexServiceSafe(request.index()).shardSafe(request.shardId());
135            Engine.Searcher searcher = indexShard.engine().acquireSearcher("termlist");
136            try {
137                Map<String,TermInfo> map = new CompactHashMap<String,TermInfo>();
138                IndexReader reader = searcher.reader();
139                Fields fields = MultiFields.getFields(reader);
140                if (fields != null) {
141                    for (String field : fields) {
142                        // skip internal fields
143                        if (field.charAt(0) == '_') {
144                            continue;
145                        }
146                        if (request.getField() == null || field.equals(request.getField())) {
147                            Terms terms = fields.terms(field);
148                            if (terms != null) {
149                                TermsEnum termsEnum = terms.iterator(null);
150                                BytesRef text;
151                                while ((text = termsEnum.next()) != null) {
152                                    // skip invalid terms
153                                    if (termsEnum.docFreq() < 1) {
154                                        continue;
155                                    }
156                                    if (termsEnum.totalTermFreq() < 1) {
157                                        continue;
158                                    }
159                                    TermInfo t = new TermInfo();
160                                    if (request.getWithDocFreq()) {
161                                        t.docfreq(termsEnum.docFreq());
162                                    }
163                                    if (request.getWithTotalFreq()) {
164                                        t.totalFreq(termsEnum.totalTermFreq());
165                                    }
166                                    map.put(text.utf8ToString(), t);
167                                }
168                            }
169                        }
170                    }
171                }
172                return new ShardTermlistResponse(request.index(), request.shardId(), map);
173            } catch (IOException ex) {
174                throw new ElasticsearchException(ex.getMessage(), ex);
175            } finally {
176                searcher.release();
177            }
178        }

180        private void merge(Map<String,TermInfo> map, Map<String,TermInfo> other) {
181            for (Map.Entry<String,TermInfo> t : other.entrySet()) {
182                if (map.containsKey(t.getKey())) {
183                    TermInfo info = map.get(t.getKey());
184                    Integer docFreq = info.getDocFreq();
185                    if (docFreq != null) {
186                        if (t.getValue().getDocFreq() != null) {
187                            info.docfreq(docFreq + t.getValue().getDocFreq());
188                        }
189                    } else {
190                        if (t.getValue().getDocFreq() != null) {
191                            info.docfreq(t.getValue().getDocFreq());
192                        }
193                    }
194                    Long totalFreq = info.getTotalFreq();
195                    if (totalFreq != null) {
196                        if (t.getValue().getTotalFreq() != null) {
197                            info.totalFreq(totalFreq + t.getValue().getTotalFreq());
198                        }
199                    } else {
200                        if (t.getValue().getTotalFreq() != null) {
201                            info.totalFreq(t.getValue().getTotalFreq());
202                        }
203                    }
204                } else {
205                    map.put(t.getKey(), t.getValue());
206                }
207            }
208        }

210        private SortedMap<String,TermInfo> sortTotalFreq(final Map<String,TermInfo> map, Integer size) {
211            Comparator<String> comp = new Comparator<String>() {
212                @Override
213                public int compare(String t1, String t2) {
214                    Long l1 = map.get(t1).getTotalFreq();
215                    String s1 = Long.toString(l1).length() + Long.toString(l1) + t1;
216                    Long l2 = map.get(t2).getTotalFreq();
217                    String s2 = Long.toString(l2).length() + Long.toString(l2) + t2;
218                    return -s1.compareTo(s2);
219                }
220            };
221            TreeMap<String,TermInfo> m = new TreeMap<String,TermInfo>(comp);
222            m.putAll(map);
223            if (size != null && size > 0) {
224                TreeMap<String,TermInfo> n = new TreeMap<String,TermInfo>(comp);
225                Map.Entry<String,TermInfo> me = m.pollFirstEntry();
226                while (me != null && size-- > 0) {
227                    n.put(me.getKey(), me.getValue());
228                    me = m.pollFirstEntry();
229                }
230                return n;
231            }
232            return m;
233        }
234    
235        private SortedMap<String,TermInfo> sortDocFreq(final Map<String,TermInfo> map, Integer size) {
236            Comparator<String> comp = new Comparator<String>() {
237                @Override
238                public int compare(String t1, String t2) {
239                    Integer i1 = map.get(t1).getDocFreq();
240                    String s1 = Integer.toString(i1).length() + Integer.toString(i1) + t1;
241                    Integer i2 = map.get(t2).getDocFreq();
242                    String s2 = Integer.toString(i2).length() + Integer.toString(i2) + t2;
243                    return -s1.compareTo(s2);
244                }
245            };
246            TreeMap<String,TermInfo> m = new TreeMap<String,TermInfo>(comp);
247            m.putAll(map);
248            if (size != null && size > 0) {
249                TreeMap<String,TermInfo> n = new TreeMap<String,TermInfo>(comp);
250                Map.Entry<String,TermInfo> me = m.pollFirstEntry();
251                while (me != null && size-- > 0) {
252                    n.put(me.getKey(), me.getValue());
253                    me = m.pollFirstEntry();
254                }
255                return n;
256            }
257            return m;
258        }
259    
260        private Map<String,TermInfo> truncate(Map<String,TermInfo> source, Integer max) {
261            if (max == null || max < 1) {
262                return source;
263            }
264            int count = 0;
265            Map<String,TermInfo> target = new CompactHashMap<String,TermInfo>();
266            for (Map.Entry<String,TermInfo> entry : source.entrySet()) {
267                if (count >= max) {
268                    break;
269                }
270                target.put(entry.getKey(), entry.getValue());
271                count++;
272            }
273            return target;
274        }

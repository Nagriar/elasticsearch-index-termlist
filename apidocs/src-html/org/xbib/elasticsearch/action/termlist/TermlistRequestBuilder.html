002    package org.xbib.elasticsearch.action.termlist;
003    
004    import org.elasticsearch.action.ActionListener;
005    import org.elasticsearch.action.support.broadcast.BroadcastOperationRequestBuilder;
006    import org.elasticsearch.client.Client;
007    import org.elasticsearch.client.internal.InternalGenericClient;
008    
009    /**
010     * A request to get termlists of one or more indices.
011     */
012    public class TermlistRequestBuilder extends BroadcastOperationRequestBuilder<TermlistRequest, TermlistResponse, TermlistRequestBuilder> {
013    
014        public TermlistRequestBuilder(InternalGenericClient client) {
015            super(client, new TermlistRequest());
016        }
017    
018        public TermlistRequestBuilder withDocFreq() {
019            request.setWithDocFreq(true);
020            return this;
021        }
022    
023        public TermlistRequestBuilder withTotalFreq() {
024            request.setWithTotalFreq(true);
025            return this;
026        }
027    
028        @Override
029        protected void doExecute(ActionListener<TermlistResponse> listener) {
030            ((Client) client).execute(TermlistAction.INSTANCE, request, listener);
031        }
032    }

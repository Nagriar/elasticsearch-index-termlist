001    
002    package org.xbib.elasticsearch.action.termlist;
003    
004    import org.elasticsearch.action.support.broadcast.BroadcastOperationRequest;
005    import org.elasticsearch.action.support.broadcast.BroadcastOperationThreading;
006    import org.elasticsearch.common.io.stream.StreamInput;
007    import org.elasticsearch.common.io.stream.StreamOutput;
008    
009    import java.io.IOException;
010    
011    public class TermlistRequest extends BroadcastOperationRequest<TermlistRequest> {
012    
013        private String field;
014    
015        private Integer size;
016    
017        private boolean withDocFreq;
018    
019        private boolean withTotalFreq;
020    
021        TermlistRequest() {
022        }
023    
024        public TermlistRequest(String... indices) {
025            super(indices);
026            operationThreading(BroadcastOperationThreading.THREAD_PER_SHARD);
027        }
028    
029        public void setField(String field) {
030            this.field = field;
031        }
032    
033        public String getField() {
034            return field;
035        }
036    
037        public void setSize(Integer size) {
038            this.size = size;
039        }
040    
041        public Integer getSize() {
042            return size;
043        }
044    
045        public void setWithDocFreq(boolean withDocFreq) {
046            this.withDocFreq = withDocFreq;
047        }
048    
049        public boolean getWithDocFreq() {
050            return withDocFreq;
051        }
052        public void setWithTotalFreq(boolean withTotalFreq) {
053            this.withTotalFreq = withTotalFreq;
054        }
055    
056        public boolean getWithTotalFreq() {
057            return withTotalFreq;
058        }
059    
060        @Override
061        public void readFrom(StreamInput in) throws IOException {
062            super.readFrom(in);
063            field = in.readString();
064        }
065    
066        @Override
067        public void writeTo(StreamOutput out) throws IOException {
068            super.writeTo(out);
069            out.writeString(field);
070        }
071    }

002    package org.xbib.elasticsearch.action.termlist;
003    
004    import java.util.AbstractSet;
005    import java.util.ConcurrentModificationException;
006    import java.util.Iterator;
007    import java.util.NoSuchElementException;
008    import java.util.Set;
009    
010    /**
011     * CompactHashSet - implementation taken from
012     * http://code.google.com/p/ontopia/source/browse/trunk/ontopia/src/java/net/ontopia/utils/CompactHashSet.java?r=1704
013     * (License Apache 2.0)
014     * load factor increased to 0.9
015     */
016    public class CompactHashSet<E> extends AbstractSet<E>
017        implements Set<E> {
018    
019        private final static int INITIAL_SIZE = 3;
020    
021        private final static double LOAD_FACTOR = 0.9;
022    
023        private final E nullObject = (E) new Object();
024    
025        private final E deletedObject = (E) new Object();
026    
027        private int elements;
028    
029        private int freecells;
030    
031        private E[] objects;
032    
033        private int modCount;
034    
035        public CompactHashSet() {
036            objects = (E[]) new Object[INITIAL_SIZE];
037            elements = 0;
038            freecells = objects.length;
039            modCount = 0;
040        }
041    
042        public CompactHashSet(int size) {
043            objects = (E[]) new Object[(size == 0 ? 1 : size)];
044            elements = 0;
045            freecells = objects.length;
046            modCount = 0;
047        }
048    
049        public Iterator<E> iterator() {
050            return new CompactHashIterator();
051        }
052    
053        public int size() {
054            return elements;
055        }
056    
057        @Override
058        public boolean isEmpty() {
059            return elements == 0;
060        }
061    
062        @Override
063        public boolean contains(Object o) {
064            if (o == null) {
065                o = nullObject;
066            }
067            int hash = o.hashCode();
068            int index = (hash & 0x7FFFFFFF) % objects.length;
069            int offset = 1;
070            while (objects[index] != null
071                    && !(objects[index].hashCode() == hash
072                    && objects[index].equals(o))) {
073                index = ((index + offset) & 0x7FFFFFFF) % objects.length;
074                offset = offset * 2 + 1;
075    
076                if (offset == -1) {
077                    offset = 2;
078                }
079            }
080            return objects[index] != null;
081        }
082    
083        @Override
084        public boolean add(E e) {
085            if (e == null) {
086                e = nullObject;
087            }
088            int hash = e.hashCode();
089            int index = (hash & 0x7FFFFFFF) % objects.length;
090            int offset = 1;
091            int deletedix = -1;
092            while (objects[index] != null
093                    && !(objects[index].hashCode() == hash
094                    && objects[index].equals(e))) {
095                if (objects[index] == deletedObject) {
096                    deletedix = index;
097                }
098                index = ((index + offset) & 0x7FFFFFFF) % objects.length;
099                offset = offset * 2 + 1;
100                if (offset == -1) {
101                    offset = 2;
102                }
103            }
104            if (objects[index] == null) {
105                if (deletedix != -1) {
106                    index = deletedix;
107                } else {
108                    freecells--;
109                }
110                modCount++;
111                elements++;
112                objects[index] = e;
113                if (1 - (freecells / (double) objects.length) > LOAD_FACTOR) {
114                    rehash(objects.length);
115                    if (1 - (freecells / (double) objects.length) > LOAD_FACTOR) {
116                        rehash(objects.length * 2 + 1);
117                    }
118                }
119                return true;
120            } else {
121                return false;
122            }
123        }
124    
125        @Override
126        public boolean remove(Object o) {
127            if (o == null) {
128                o = nullObject;
129            }
130            int hash = o.hashCode();
131            int index = (hash & 0x7FFFFFFF) % objects.length;
132            int offset = 1;
133            while (objects[index] != null
134                    && !(objects[index].hashCode() == hash
135                    && objects[index].equals(o))) {
136                index = ((index + offset) & 0x7FFFFFFF) % objects.length;
137                offset = offset * 2 + 1;
138                if (offset == -1) {
139                    offset = 2;
140                }
141            }
142            if (objects[index] != null) {
143                objects[index] = deletedObject;
144                modCount++;
145                elements--;
146                return true;
147            } else {
148                return false;
149            }
150        }
151    
152        @Override
153        public void clear() {
154            elements = 0;
155            for (int ix = 0; ix < objects.length; ix++) {
156                objects[ix] = null;
157            }
158            freecells = objects.length;
159            modCount++;
160        }
161    
162        @Override
163        public Object[] toArray() {
164            Object[] result = new Object[elements];
165            Object[] obj = this.objects;
166            int pos = 0;
167            for (int i = 0; i < obj.length; i++) {
168                if (obj[i] != null && obj[i] != deletedObject) {
169                    if (obj[i] == nullObject) {
170                        result[pos++] = null;
171                    } else {
172                        result[pos++] = obj[i];
173                    }
174                }
175            }
176            return result;
177        }
178    
179        @Override
180        public Object[] toArray(Object a[]) {
181            int size = elements;
182            if (a.length < size) {
183                a = (Object[]) java.lang.reflect.Array.newInstance(
184                        a.getClass().getComponentType(), size);
185            }
186            Object[] obj = this.objects;
187            int pos = 0;
188            for (int i = 0; i < obj.length; i++) {
189                if (obj[i] != null && obj[i] != deletedObject) {
190                    if (obj[i] == nullObject) {
191                        a[pos++] = null;
192                    } else {
193                        a[pos++] = obj[i];
194                    }
195                }
196            }
197            return a;
198        }
199    
200        protected void rehash(int newCapacity) {
201            Object[] newObjects = new Object[newCapacity];
202            for (Object o : objects) {
203                if (o == null || o == deletedObject) {
204                    continue;
205                }
206                int hash = o.hashCode();
207                int index = (hash & 0x7FFFFFFF) % newCapacity;
208                int offset = 1;
209                while (newObjects[index] != null) {
210                    index = ((index + offset) & 0x7FFFFFFF) % newCapacity;
211                    offset = offset * 2 + 1;
212    
213                    if (offset == -1) {
214                        offset = 2;
215                    }
216                }
217    
218                newObjects[index] = o;
219            }
220            objects = (E[]) newObjects;
221            freecells = objects.length - elements;
222        }
223    
224        private class CompactHashIterator implements Iterator<E> {
225    
226            private int index;
227            private int lastReturned = -1;
228            private int expectedModCount;
229    
230            public CompactHashIterator() {
231                for (index = 0; index < objects.length
232                        && (objects[index] == null
233                        || objects[index] == deletedObject); index++) {
234                    ;
235                }
236                expectedModCount = modCount;
237            }
238    
239            public boolean hasNext() {
240                return index < objects.length;
241            }
242    
243            public E next() {
244                if (modCount != expectedModCount) {
245                    throw new ConcurrentModificationException();
246                }
247                int length = objects.length;
248                if (index >= length) {
249                    lastReturned = -2;
250                    throw new NoSuchElementException();
251                }
252    
253                lastReturned = index;
254                for (index += 1; index < length
255                        && (objects[index] == null
256                        || objects[index] == deletedObject); index++) {
257                    ;
258                }
259                if (objects[lastReturned] == nullObject) {
260                    return null;
261                } else {
262                    return objects[lastReturned];
263                }
264            }
265    
266            public void remove() {
267                if (modCount != expectedModCount) {
268                    throw new ConcurrentModificationException();
269                }
270                if (lastReturned == -1 || lastReturned == -2) {
271                    throw new IllegalStateException();
272                }
273                if (objects[lastReturned] != null && objects[lastReturned] != deletedObject) {
274                    objects[lastReturned] = deletedObject;
275                    elements--;
276                    modCount++;
277                    expectedModCount = modCount;
278                }
279            }
280        }

001    package org.xbib.elasticsearch.action.termlist;
002    
003    import org.elasticsearch.common.io.stream.StreamInput;
004    import org.elasticsearch.common.io.stream.StreamOutput;
005    import org.elasticsearch.common.io.stream.Streamable;
006    
007    import java.io.IOException;
008    
009    public class TermInfo implements Streamable {
010    
011        Integer docFreq;
012    
013        Long totalFreq;
014    
015        public TermInfo docfreq(int docFreq) {
016            this.docFreq = docFreq;
017            return this;
018        }
019    
020        public Integer getDocFreq() {
021            return docFreq;
022        }
023    
024        public TermInfo totalFreq(long totalFreq) {
025            this.totalFreq = totalFreq;
026            return this;
027        }
028    
029        public Long getTotalFreq() {
030            return totalFreq;
031        }
032    
033        @Override
034        public void readFrom(StreamInput in) throws IOException {
035            boolean b = in.readBoolean();
036            if (b) {
037                docfreq(in.readInt());
038            }
039            b = in.readBoolean();
040            if (b) {
041                totalFreq(in.readVLong());
042            }
043        }
044    
045        @Override
046        public void writeTo(StreamOutput out) throws IOException {
047            Integer i = getDocFreq();
048            if (i != null) {
049                out.writeBoolean(true);
050                out.writeInt(i);
051            } else {
052                out.writeBoolean(false);
053            }
054            Long l = getTotalFreq();
055            if (l != null) {
056                out.writeBoolean(true);
057                out.writeVLong(l);
058            } else {
059                out.writeBoolean(false);
060            }
061        }
062    }

002package org.xbib.elasticsearch.rest.action.termlist;
003
004import java.io.IOException;
005import java.util.Map;
006
007import org.elasticsearch.action.ActionListener;
008import org.elasticsearch.client.Client;
009import org.elasticsearch.common.Strings;
010import org.elasticsearch.common.inject.Inject;
011import org.elasticsearch.common.settings.Settings;
012import org.elasticsearch.common.xcontent.XContentBuilder;
013import org.elasticsearch.rest.BaseRestHandler;
014import org.elasticsearch.rest.RestChannel;
015import org.elasticsearch.rest.RestController;
016import org.elasticsearch.rest.RestRequest;
017import org.elasticsearch.rest.XContentRestResponse;
018import org.elasticsearch.rest.XContentThrowableRestResponse;
019import org.elasticsearch.rest.action.support.RestXContentBuilder;
020
021import org.xbib.elasticsearch.action.termlist.TermInfo;
022import org.xbib.elasticsearch.action.termlist.TermlistAction;
023import org.xbib.elasticsearch.action.termlist.TermlistRequest;
024import org.xbib.elasticsearch.action.termlist.TermlistResponse;
025
026import static org.elasticsearch.rest.RestRequest.Method.GET;
027import static org.elasticsearch.rest.RestStatus.OK;
028import static org.elasticsearch.rest.action.support.RestActions.buildBroadcastShardsHeader;
029
030public class RestTermlistAction extends BaseRestHandler {
031
032    @Inject
033    public RestTermlistAction(Settings settings, Client client, RestController controller) {
034        super(settings, client);
035        controller.registerHandler(GET, "/_termlist", this);
036        controller.registerHandler(GET, "/{index}/_termlist", this);
037        controller.registerHandler(GET, "/{index}/{field}/_termlist", this);
038    }
039
040    public void handleRequest(final RestRequest request, final RestChannel channel) {
041        TermlistRequest termlistRequest = new TermlistRequest(
042                Strings.splitStringByCommaToArray(request.param("index")));
043        termlistRequest.setField(request.param("field"));
044        termlistRequest.setSize(request.paramAsInt("size", 0));
045        termlistRequest.setWithDocFreq(request.paramAsBoolean("docfreqs", false));
046        termlistRequest.setWithTotalFreq(request.paramAsBoolean("totalfreqs", false));
047        client.execute(TermlistAction.INSTANCE, termlistRequest, new ActionListener<TermlistResponse>() {
048
049            public void onResponse(TermlistResponse response) {
050                try {
051                    XContentBuilder builder = RestXContentBuilder.restContentBuilder(request);
052                    builder.startObject();
053                    buildBroadcastShardsHeader(builder, response);
054                    builder.startArray("terms");
055                    for (Map.Entry<String,TermInfo> t : response.getTermlist().entrySet()) {
056                        builder.startObject().field("name", t.getKey());
057                        if (t.getValue().getDocFreq() != null) {
058                            builder.field("docfreq", t.getValue().getDocFreq());
059                        }
060                        if (t.getValue().getTotalFreq() != null) {
061                            builder.field("totalfreq", t.getValue().getTotalFreq());
062                        }
063                        builder.endObject();
064                    }
065                    builder.endArray().endObject();
066                    channel.sendResponse(new XContentRestResponse(request, OK, builder));
067                } catch (Exception e) {
068                    onFailure(e);
069                }
070            }
071
072            public void onFailure(Throwable e) {
073                try {
074                    channel.sendResponse(new XContentThrowableRestResponse(request, e));
075                } catch (IOException e1) {
076                    logger.error("Failed to send failure response", e1);
077                }
078            }
079        });
080    }
081}

001    package org.xbib.elasticsearch.plugin.termlist;
002    
003    import org.elasticsearch.action.ActionModule;
004    import org.xbib.elasticsearch.action.termlist.TermlistAction;
005    import org.xbib.elasticsearch.action.termlist.TransportTermlistAction;
006    import org.elasticsearch.plugins.AbstractPlugin;
007    import org.elasticsearch.rest.RestModule;
008    import org.xbib.elasticsearch.rest.action.termlist.RestTermlistAction;
009    
010    public class IndexTermlistPlugin extends AbstractPlugin {
011    
012        public String name() {
013            return "index-termlist";
014        }
015    
016        public String description() {
017            return "Index termlists for Elasticsearch";
018        }
019    
020        public void onModule(RestModule module) {
021            module.addRestAction(RestTermlistAction.class);
022        }
023    
024        public void onModule(ActionModule module) {
025            module.registerAction(TermlistAction.INSTANCE, TransportTermlistAction.class);
026        }
027    
028    }

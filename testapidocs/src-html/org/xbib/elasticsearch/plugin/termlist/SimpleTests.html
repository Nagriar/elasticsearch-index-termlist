001    package org.xbib.elasticsearch.plugin.termlist;
002    
003    import org.elasticsearch.action.admin.cluster.node.info.NodesInfoResponse;
004    import org.elasticsearch.client.Client;
005    import org.elasticsearch.common.logging.ESLogger;
006    import org.elasticsearch.common.logging.ESLoggerFactory;
007    import org.elasticsearch.common.network.NetworkUtils;
008    import org.elasticsearch.common.settings.Settings;
009    import org.elasticsearch.node.Node;
010    import org.testng.Assert;
011    import org.testng.annotations.AfterClass;
012    import org.testng.annotations.BeforeClass;
013    import org.testng.annotations.Test;
014    
015    import static org.elasticsearch.common.settings.ImmutableSettings.settingsBuilder;
016    import static org.elasticsearch.node.NodeBuilder.nodeBuilder;
017    
018    public class SimpleTests extends Assert {
019    
020        private final static ESLogger logger = ESLoggerFactory.getLogger("test");
021    
022        protected final String CLUSTER = "test-cluster-" + NetworkUtils.getLocalAddress().getHostName();
023    
024        private Node node;
025    
026        private Client client;
027    
028        @BeforeClass
029        public void startNode() {
030            Settings finalSettings = settingsBuilder()
031                    .put("cluster.name", CLUSTER)
032                    .put("discovery.zen.ping.multicast.enabled", false)
033                    .put("node.local", true)
034                    .put("gateway.type", "none")
035                    .build();
036            node = nodeBuilder().settings(finalSettings).build().start();
037            client = node.client();
038        }
039    
040        @AfterClass
041        public void stopNode() {
042            node.close();
043        }
044    
045        @Test
046        public void assertPluginLoaded() {
047            NodesInfoResponse nodesInfoResponse = client.admin().cluster().prepareNodesInfo()
048                    .clear().setPlugin(true).get();
049            logger.info("{}", nodesInfoResponse);
050            assertEquals(nodesInfoResponse.getNodes().length, 1);
051            assertNotNull(nodesInfoResponse.getNodes()[0].getPlugins().getInfos());
052            assertEquals(nodesInfoResponse.getNodes()[0].getPlugins().getInfos().size(), 1);
053            assertEquals(nodesInfoResponse.getNodes()[0].getPlugins().getInfos().get(0).getName(), "index-termlist");
054            assertEquals(nodesInfoResponse.getNodes()[0].getPlugins().getInfos().get(0).isSite(), false);
055        }
056    }
